# Create ssh file so you can login. Do this on new sd card boot volume.
# This is a security feature of raspi
touch /Volumes/boot/ssh

# wpa supplicant to learn your network. Format file as below
# check ssh touch step if you have since booted
vim /Volumes/boot/wpa_supplicant.conf

ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
country=<Insert 2 letter ISO 3166-1 country code here>
update_config=1

network={
 ssid="<Name of your wireless LAN>"
 psk="<Password for your wireless LAN>"
}
